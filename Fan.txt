package testPro;public class Fan {	private int currentSpeed = 0;	private boolean state = true;	public int increaseSpeed() {		currentSpeed = currentSpeed == 3 ? currentSpeed = 0 : ++currentSpeed;		System.out.format("Current Speed: %d, Direction: %s \n", currentSpeed,				state == true ? "Clockwise" : "Counterclockwise");		return currentSpeed;	}	public void reverseDirection() throws Exception {		if (this.currentSpeed != 0)			throw new Exception("Please turn off the fan before changing the direction.");		state = !state;		System.out.println("Change direction");	}}
